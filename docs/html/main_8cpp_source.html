25 std::string get_last_line(std::fstream &fin)
26 {
27  if(fin.is_open()) {
28  fin.seekg(-2, std::ios_base::end); // go to one spot before the EOF
29  bool keepLooping = true;
30  while(keepLooping) {
31  char ch;
32  fin.get(ch); // Get current byte's data
33 
34  if((int)fin.tellg() <= 1) { // If the data was at or before the 0th byte
35  fin.seekg(0); // The first line is the last line
36  keepLooping = false; // So stop there
37  }
38  else if(ch == '\n') { // If the data was a newline
39  keepLooping = false; // Stop at the current position.
40  }
41  else { // If the data was neither a newline nor at the 0 byte
42  fin.seekg(-2, std::ios_base::cur); // Move to the front of that data, then to the front of the data before it
43  }
44  }
45 
46  std::string lastLine;
47  getline(fin,lastLine);
48  return lastLine; // Display it
49  }
50  return "";
51 }
