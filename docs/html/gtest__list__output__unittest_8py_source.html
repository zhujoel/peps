248  'test_out.' + out_format)

252  command = ([

257  environ_copy = os.environ.copy()

260 
261  self.assertTrue(p.exited)
262  self.assertEqual(0, p.exit_code)
263  self.assertTrue(os.path.isfile(file_path))
264  with open(file_path) as f:
